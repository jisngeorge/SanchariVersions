```json
{
  "timetable": {
    "version": 102,
    "url": "https://github.com/your-username/your-repo/releases/download/v1.2/TimetableDatabase.db"
  },
  "community": {
    "version": 451,
    "url": "https://github.com/your-username/your-repo/releases/download/v1.2/CommunityDatabase.db"
  },
  "app": {
    "versionCode": 2,
    "url": "https://github.com/your-username/your-repo/releases/download/v1.2/app-release.apk",
    "mandatory": false
  },
  "versions": "https://your-username.github.io/your-repo/version.json",
  "communityData": "https://script.google.com/macros/s/YOUR_SCRIPT_ID/exec"
}
```
'versions' and 'communityData' are optional.
